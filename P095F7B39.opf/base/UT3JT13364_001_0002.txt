བསྡུ་སྒྲིག་གསལ་བཤད།

 ད་ལམ་ང་ཚོའི་ཇོ་ནང་རྗེ་བཙུན་ཏཱ་ར་ནཱ་ཐའི་གསུང་འབུམ་ཕྱོགས་གཅིག་ཏུ་བསྒྲིགས་ཏེ་པར་སྐྲུན་ཞུས་པ་འདི་ནི། གཟན་དཀར་མཆོག་སྤྲུལ་ཐུབ་བསྟན་ཉི་མའི་བཀའ་སློབ་དང་བདག་རྐྱེན་འོག་བྱུང་བ་ཞིག་ཡིན། དེ་ཡང་སྤྱིར་ཁོང་ལ་གསུང་འབུམ་པོད་བཞི་བཅུ་ཙམ་ཞིག་ཡོད་ཚུལ་ཕྱིས་སུ་བྲིས་པའི་རྣམ་ཐར་འགའ་ལས་གསལ་ཡང༌། དངོས་སུ་པོད་བཞི་བཅུའི་དཀར་ཆག་ཁ་གསལ་ཞིག་ད་སྔོན་ང་ཚོའི་མཇལ་རྒྱུ་མ་བྱུང༌། ད་ལམ་དངོས་སུ་མཇལ་རྒྱུ་བྱུང་བའི་པོད་བཅོ་ལྔ་ནི། དེང་སང་ཡོངས་གྲགས་སུ་རྟག་བརྟན་པར་མ་ཞེས་འབོད་པ་འདི་ནི། བོད་རབ་བྱུང་བཅུ་པའི་མེ་འབྲུག （ སྤྱི་ལོ 1616） ལ། མངའ་རིས་སྨད་ཀྱི་གུང་ཐང་གི་རྒྱལ་རབས་ཉི་ཤུ་རྩ་གསུམ་པའམ། ཡང་ལུགས་གཅིག་ལ་དམར་པོ་སྡེ་བསྒྲངས་ན་ཉི་ཤུ་རྩ་བཞི་པ་བསོད་ནམས་དབང་ཕྱུག་སྡེས་ཡོན་བདག་བྱས། ཏཱ་ར་ནཱ་ཐའི་དངོས་སློབ་གཉའ་ཡི་བནྡེ་དཀོན་མཆོག་ལེགས་པས། ཐོག་མའི་མཚམས་སྦྱོར་དང་པར་བྱང་འབྲི་བ་སོགས་བྱས་ཏེ། དུས་འཁོར་སྔགས་ཕྲེང་དང་སྐད་གཉིས་ཤན་སྦྱར་སོགས་ཆོས་ཚན་བཅུ་བཞི་ཤིང་པར་དུ་བཀོས་པ། རྟག་བརྟན་དམ་ཆོས་གླིང་དུ་སྤྱན་དྲངས་ཏེ་བཞུགས་སུ་གསོལ། དེའི་སྟེང་དུ་རྟག་བརྟན་རང་ནས་བརྐོས་པ་བཅས་པོད་བཅོ་ལྔ་ཙམ་ཞིག་ཡོད་པ་དེའི་བུ་དཔེ་པར་མ་ཚར་གཅིག །རྒྱལ་དབང་ལྔ་པ་ཆེན་པོས་འབྲས་སྤུངས་དཔེ་མཛོད་ཁང་དུ་གདན་དྲངས་བ། ཕྱིས་དུས་འགྱུར་སྐབས་སུ་པེ་ཅིང་མི་རིགས་ཕོ་བྲང་དུ་བཞུགས་ཀྱང༌། དེང་སེང་བོད་ལྗོངས་དཔེ་མཛོད་ཁང་དུ་སྤྱན་དྲངས་ཏེ་བཞུགས་པ་དེ་ཡིན། ཡང་གངས་ཅན་མཁས་གྲུབ་རིམ་བྱོན་གྱི་མིང་མཛོད་ལས། ཁོང་ལ་གསུང་རྩོམ་ཤིན་ཏུ་མང་པོ་ཡོད་པར་གྲགས་ཀྱང་རྒྱལ་དབང་ལྔ་པ་ཆེན་པོས། གསུང་འབུམ་ཤིང་པར་རྣམས་བཀའ་རྒྱ་བཏབ་སྟེ་པར་མི་ཆོག་པ་མཛད་པས་ད་ལྟ་ཁོང་གི་གསུང་རྩོམ་རྣམས་ཁ་ཚང་བ་ཧ་ཅང་དཀའ། རྒྱལ་དབང་ལྔ་པ་ཞིང་དུ་གཤེགས་རྗེས། བློ་གྲོས་བརྟན་པས་ཁོང་གི་གསུང་འབུམ་ལ་དཀར་ཆག་བསྡུས་པ་ཞིག་མཛད་འདུག

།དེའི་རྗེས་ཞ་ལུ་རི་ཕུག་སྤྲུལ་སྐུས་མཛད་པའི་ཏཱ་ར་ནཱ་ཐའི་གསུང་འབུམ་དཀར་ཆག་དྭངས་གསལ་མེ་ལོང་ཞེས་པའི་ནང་དུ། པོད་ཀ་ནས་ཙ་བར་བཅུ་བདུན་ལ་ཆོས་ཚན་ཉིས་བརྒྱ་དོན་གཉིས་དང༌། ལྡེབ་གྲངས་བདུན་སྟོང་བརྒྱད་བརྒྱ་ལྔ་བཅུ་བཞུགས། ད་ལྟ་པེ་ཅིང་མི་རིགས་ཕོ་བྲང་དུ་པོད་བཅུ་བདུན་ལ་ཆོས་ཚན་ཉིས་བརྒྱ་དོན་བརྒྱད་དང༌། ལྡེབ་གྲངས་བདུན་སྟོང་དགུ་བརྒྱ་ལྔ་བཅུ་རྩ་བརྒྱད་བཞུགས་ཞེས་གསལ་ཡང་ང་ཚོའི་གོང་གི་དཀར་ཆག་གཉིས་པོ་ད་སྔོན་མཇལ་རྒྱུ་མ་རྙེད། དེ་ཡང་ཞ་ལུ་རི་ཕུག་སྤྲུལ་སྐུས་དཀར་ཆག་དྭངས་གསལ་མེ་ལོང་ཞེས་པ་ལྗགས་རྩོམ་གནང་སྟེ། སྔར་རྟག་བརྟན་གྱི་ཤིང་པར་དུས་ཟིང་དབང་གིས་ཉམས་ཆག་ཤོར་བ་རྣམས། ཁ་གསོ་མཛད་ཚུལ་ཐོས་ཀྱང་ཞིབ་ཕྲའི་ལོ་རྒྱུས་ཁ་གསལ་མཇལ་རྒྱུ་མ་བྱུང༌། ད་ལྟ་དགེན་འཇམ་དབྱངས་རྣམས་རྒྱལ་གྱི་དཔེ་མཛོད་དུ་བཞུགས་པའི་པར་རྙིང་པོད་བཅུ་བདུན་འདི། ཞ་ལུ་རི་ཕུག་སྤྲུལ་སྐུས་ཉམས་གསོ་མཛད་པ་དེ་ཡིན་ནམ་སྙམས་སྟེ། ད་ལྟ་རྟག་བརྟན་གྱི་པར་རྙིང་བོད་ལྗོངས་དཔེ་མཛོད་ཁང་དུ་བཞུགས་པ་པོད་བཅོ་ལྔ་དང༌། པོད་གྲངས་དང་ཆོས་ཚན་མང་ཉུང་ཚིག་མཚམས་སོགས་ཅུང་མི་འདྲ་བ་ཡོད་པའི་ཕྱིར། ད་ལམ་ང་ཚོས་ཁོང་གི་གསུང་འབུམ་གློག་ཀླད་ནང་དུ་བཅུག་སྐབས་ཀྱི་ཐོག་མའི་མ་དཔེ་ནི། ཉེ་ཆར་བླ་མ་ཡོན་ཏན་བཟང་པོས་འཛམ་ཐང་དུ་ཤིང་པར་ལ་བརྐོས་པའི་གླེགས་བམ་ཉེར་བཞི་ཅན་དེ་ལ་བྱས་ཏེ་ཞུ་དག་ཐེངས་གསུམ་བྱས། དེ་རྗེས་སྔར་ཇོ་ནང་རྟག་བརྟན་དུ་བཞུགས་པ་རྩོམ་པ་པོ་ཏཱ་ར་ནཱ་ཐ་རང་ཉིད་ཀྱི་ཕྱག་དཔེ་སྐབས་འགར་ཕྱག་མཆན་ཡོད་པ། ད་ལྟ་འབྲས་སྤུངས་གནས་བཅུ་ལྷ་ཁང་ནང་བཞུགས་བྲིས་མ་པོད་ཆུང་བཅུ་ལ་ཆོས་ཚན༤༣ལ་བསྡུར་གཞི་དང༌། ཇོ་ནང་རྟག་བརྟན་གྱི་པར་རྙིང་པོད་བཅོ་ལྔ་བོད་ལྗོངས་དཔེ་མཛོད་ཁང་དུ་བཞུགས་པ་དང༌། ཨ་རི་བ་དགེན་འཇམ་དབྱངས་རྣམ་རྒྱལ་གྱི་དཔེ་མཛོད་དུུ་བཞུགས་པ་པར་རྙིང་པོད་བཅུ་བདུན་ཡོད་པ་གཉིས་ཕྱོགས་བསྡོམས་ཀྱིས།

པར་རྙིང་པོད་ཉི་ཤུ་ཙམ་ལ་ཆོས་ཚན་༣༡༤བྱུང་བ་ལ་སྡུར་བྱེད་བྱས་ཏེ་དཔེ་བསྡུར་ཐེངས་དང་པོ་ཞུས། དེ་ནས་རྟག་བརྟན་གྱི་པར་པོད་ཉི་ཤུ་ལ་བསྡུར་གཞི་དང༌། འཛམ་ཐང་གི་པར་མས་སྡུར་བྱེད་བྱས་ཏེ་དཔེ་བསྡུར་ཐེང་གཉིས་པ་ཞུས་པ་ཡིན། འོན་ཀྱང་པར་མ་སྔ་ཕྱི་གཉིས་དང་བྲིས་མ་གཅིག་བཅས་དཔེ་གསུམ་པོ་རྩ་བའི་ཆ་ནས་གཅིག་མཚུངས་སུ་བཞུགས་པ་ལས་ཁྱད་པར་ཆེན་པོ་མི་འདུག །ཆོ་གའི་སྐོར་ལ་མི་འདྲ་བ་ཅུང་ཟད་ཡོད་པ་ནི་མཆན་དཀྱུས་སུ་ཤོར་བ་ཡིན་འདུག་པ་རྣམས་ཀྱང༌། གུག་རྟགས་ནང་དུ་བཅུག་ཏེ་གསལ་པོ་བྱས་ཡོད། རྟག་བརྟན་པར་རྙིང་དང་འབྲས་སྤུངས་བྲིས་མ་གཉིས་སུ་དཔེ་མ་བཞུགས་པ། འཛམ་ཐང་པར་མའི་ནང་བཞུགས་གསུང་ཐོར་བུའི་སྐོར་ཝ་རྟགས་ཅན་དང༌། ཞ་རྟགས་ཅན་གཉིས་དང༌། ཟ་རྟགས་ཅན་དབུ་མ་ཐེག་མཆོག་གི་ཟིན་བྲིས་རབ་བྱེད་དྲུག་པ་ཡན་ཆད། ཏཱ་ར་ནཱ་ཐའི་ཞལ་གསུང་སློབ་མ་ཡེ་ཤེས་རྒྱ་མཚོས་ཟིན་བྲིས་སུ་བཀོད་པ་དང༌། རབ་བྱེད་བདུན་པ་བརྒྱད་པ་གཉིས་ཛ་འགོའི་དགེ་བཤེས་མཆོག་གྲུབ་རྒྱ་མཚོས་གསར་དུ་བསྣན་པ། འ་རྟགས་ཅན་ཉང་ཡུལ་གྱི་ལོ་རྒྱུསུ་ཏཱ་ར་ནཱ་ཐས་ཞབས་བཏགས་མཛད་པར་གྲགས་པ་བཅས་ཆོས་ཚན༡༢༡ལ་འཛམ་ཐང་གི་པར་མའི་དཔེ་གཅིག་པུ་ལས། རྟག་བརྟན་པར་རྙིང་དང་འབྲས་སྤུངས་བྲིས་མ་གཉིས་ཀྱི་བསྡུར་གཞི་མ་རྙེད། རྟག་བརྟན་པར་རྙིང་གཅིག་པུ་ལས་གཞན་གཉིས་ལ་མེད་པ་ཆོས་ཚན༤། ཡང་རྟག་བརྟན་དང་འཛམ་ཐང་པར་མ་གཉིས་ཆར་མེད་པ། འབྲས་སྤུངས་གནས་བཅུ་ལྷ་ཁང་དུ་བཞུགས་པའི་གྲལ་ནས་ཆོས་ཚན་བཅོ་བརྒྱད་ལ་ལྡེབ༡༩༣གསར་རྙེད་བྱུང་བ་བཅས་ལ། ཞུ་དག་བྱས་ཏེ་ཁ་སྐོང་གི་ཚུལ་དུ་བཞག་ཡོད་པ་བཅས་ཁྱོན་བསྡོམས་དེབ་བཞི་བཅུ་ཞེ་ལྔ་ལ་ཆོས་ཚན༩༠༠ཡོད། ད་ལྟ་ང་ཚོའི་མ་དཔེ་གཡར་རྒྱུ་མ་རྙེད་པ་སྔོན་རྩོམ་པ་པོའི་ཕྱག་དཔེའི་གྲལ་ནས་ཡིན་པ། འབྲས་སྤུངས་གནས་བཅུ་ལྷ་ཁང༌། དཔེ་རྟགས་ཕྱི༢༩༨ནང་བཞུགས་གཏམ་དུ་བྱ་བ་ནོར་བུའི་འབྱུང་གནས་ལྡེབ༦། ཡང་ཕྱི༥༠༨ནང་བཞུགས་འཁོར་ལོ་གསུམ་གྱི་རྣམ་བཞག་ཆོས་དབྱིངས་རྒྱ་མཚོའི་རྒྱན་རྣམ་པར་བཀོད་པ་ལྡེབ༩༢།

སྟེ་ལག་བྲིས་ཀྱི་ཆོས་ཚན་གཉིས་པོ་མ་འོངས་པར་དཔེ་གཡར་ཐབས་བྱ་དགོས། ད་ལྟ་ང་ཚོས་དཔེ་བསྡུར་དང་ཞུ་དག་གི་སྐབས་སུ་མ་དཔེ་རྩ་འཛིན་བྱས་ཏེ། དཔེ་གསུམ་མི་འདྲ་བའི་ཚེ་རྟག་བརྟན་ལ <<རྟ>> རྟགས་འདི་དང༌། འཛམ་ཐང་ལ་རྟགས <<ཛཾ་>> འདི། དག་ཆ་ནོར་བཅོས་ལ་གུག་རྟག་འདི [ ] དང༌། མཆན་རྣམས་གུག་རྟག < > འདིའི་ནང་དུ་བླུག་ཡོད། གསལ་བཤད་ཞུ་རྒྱུ་ཞིག་ལ་དེབ་བཞི་པའི་ནང་དུ། ཀུན་མཁྱེན་དོལ་པོ་པའི་གསུང་ཆོས་ཚན་བཞི། རིག་འཛིན་ཚེ་དབང་ཀུན་ཁྱབ་ཀྱི་གསུང་ལྷོ་སྒོའི་ཆོ་ག་ཞེས་པའི་ཆོས་ཚན་གཅིག་བཅས་འཛམ་ཐང་པར་མར་བཞག་ཡོད་པ་དང༌། ཡང་ས་བཟང་མ་ཏི་པཎ་ཆེན་གྱི་གསུང་དུས་འཁོར་ཕྱག་རྒྱའི་སྐོར་ཆོས་ཚན་གཅིག་རྟག་བརྟན་པར་རྙིང་དུ་ཡོད་པ་རྣམས་རང་འཇག་བཞག་ཡོད་པ་དང༌། ཁོང་གི་རང་རྣམ་གཙོ་བོར་གྱུར་པའི་ལོ་རྒྱུས་ཞལ་གདམས་རྣམས་སུ། གཙང་གི་ཡུལ་སྐད་ཡིན་ནམ་སྙམ་པའི་ཐ་སྙད་གོ་དཀའ་བ་མང་པོ་ཞིག་བེད་སྤྱད་གནང་བ་དག་ཀྱང་རང་སོར་བཞག་ཡོད། ད་ལམ་བཀའ་འབུམ་འདི་རྩིས་འཁོར་ནང་དུ་བླུགས་པའི་འགྲོ་སོང་ཙཱ་འདྲ་ཚོགས་པ་ནས་ཞུས་ཤིང༌། པར་སྐྲུན་ཚར་ལྔ་བརྒྱའི་འགྲོ་སོང་བོད་རིག་པ་དཔེ་སྐྲུན་ཁང་ནས་བགྱིས། ཚར་བརྒྱད་བརྒྱའི་འགྲོ་སོང་ཧྲིན་ཀྲིན་སོགས་ཀྱི་་་་རྒྱའི་སྦྱིན་བདག་རྣམས་ཀྱིས་གནང་ཞིང༌། ཚར་ཉིས་བརྒྱའི་འགྲོ་སོང་ཙཱ་འདྲ་ཚོགས་པ་ནས་བགྱིས་པ་བཅས་ཀྱི་མཐུན་རྐྱེན་ལས། ད་ལམ་དངོས་སུུ་པར་དུ་བསྐྲུན་ཐུབ་པ་བྱུང་སྟེ་རྒྱ་ཆེའི་ཀློག་པ་པོ་རྣམས་ཀྱི་སྤྱན་ལམ་དུ་ཕུལ་བ་ཡིན། ལས་གཞི་འདིར་བོད་ལྗོངས་དཔེ་མཛོད་ཁང་ནས་མ་དཔེ་པོད་བཅོ་ལྔ་གཡར་གནང་བྱུང་བ་དང༌། འཛམ་ཐང་བླ་མ་མུན་སེལ་མཆོག་ནས་ཀྱང་མ་དཔེ་བཙལ་རོགས་དང༌། གོ་རིམ་སྒྲིག་རྒྱུ། ཞུ་དག་བཅས་ལ་རོགས་རམ་གང་མང་གནང་བ་དང༌། ཁམས་སྤྲུལ་བསོད་ནམས་དོན་གྲུབ་མཆོག་ནས་ཉང་ཡུལ་ཆོས་འབྱུང་ཞུ་དག་གི་རོགས་རམ་གནང་བ་དང༌། བླ་མ་དཔལ་ལྡན་རྒྱ་མཚོས་ཞུ་དག་གི་ལས་ཀ་སྤྱིའི་ཐུགས་འགན་ཆེ་བཞེས་གནང་བ་བཅས་ལ་ཐུགས་རྗེ་ཆེ་ཞུ་རྒྱུ་བའོ།
